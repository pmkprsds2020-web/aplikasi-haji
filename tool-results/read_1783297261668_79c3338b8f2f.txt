   672→--   • Staff (super_admin/admin/kepala_klinik/pj_mutu/petugas) → full CRUD, sees non-deleted rows
   673→--   • Jamaah (patient) → read/insert own linked rows only
   674→--   • Soft delete: SELECT excludes deleted_at IS NOT NULL for non-super-admins
   675→
   676→-- ---------- PROFILES ----------
   677→alter table public.profiles enable row level security;
   678→drop policy if exists profiles_select on public.profiles;
   679→create policy profiles_select on public.profiles for select using (auth.uid() = id or public.is_staff());
   680→drop policy if exists profiles_insert on public.profiles;
   681→create policy profiles_insert on public.profiles for insert with check (auth.uid() = id or public.is_staff());
   682→drop policy if exists profiles_update on public.profiles;
   683→create policy profiles_update on public.profiles for update using (auth.uid() = id or public.is_staff());
   684→drop policy if exists profiles_delete on public.profiles;
   685→create policy profiles_delete on public.profiles for delete using (public.is_super_admin());

   691→drop policy if exists audit_log_insert on public.audit_log;
   692→create policy audit_log_insert on public.audit_log for insert with check (auth.uid() is not null);
   693→drop policy if exists audit_log_update on public.audit_log;

   701→alter table public.jamaah enable row level security;

   934→-- 10. SUPABASE REALTIME — enable broadcast for chat tables
   935→-- ============================================================================
   936→-- Adds chat_message and chat_room to the realtime publication so that
   937→-- INSERT/UPDATE/DELETE events are broadcast to subscribed clients.
   938→-- Without this, supabase.channel(...).on('postgres_changes', ...) won't fire.
   939→do $$
   940→begin
   941→  -- Add tables to the realtime publication (ignore if already present)
   942→  begin
   943→    alter publication supabase_realtime add table public.chat_message;
   944→  exception when others then null;
   945→  end;
   946→  begin
   947→    alter publication supabase_realtime add table public.chat_room;
   948→  exception when others then null;
   949→  end;
   950→end;
   951→$$;
   952→
   953→-- Also ensure the publication exists (Supabase creates it by default, but just in case)
   954→do $$
   955→begin
   956→  -- Check if publication exists; if not, this will error silently
   957→  perform 1 from pg_publication where pubname = 'supabase_realtime';
   958→  -- If we get here, publication exists — tables are already added above
   959→exception when others then
   960→  -- Publication doesn't exist — create it (rare case)
   961→  begin
   962→    create publication supabase_realtime for table public.chat_message, public.chat_room;
   963→  exception when others then null;
   964→  end;
   965→end;
   966→$$;
